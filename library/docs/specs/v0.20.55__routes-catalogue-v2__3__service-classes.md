## Section 4 — Service Classes

### 4.1 Playwright__Service (Thin Orchestrator)

The orchestrator holds references to all sub-services and delegates. **No validation logic, no execution logic, no artefact logic.** Only wiring and simple delegations.

```python
# ═══════════════════════════════════════════════════════════════════════════════
# Playwright__Service — thin orchestrator
#
# Holds references to all sub-services. Routes call methods here;
# methods delegate to the appropriate specialist class.
# No business logic lives here — just wiring.
# ═══════════════════════════════════════════════════════════════════════════════


class Playwright__Service(Type_Safe):

    capability_detector  : Capability__Detector
    session_manager      : Session__Manager
    browser_launcher     : Browser__Launcher
    request_validator    : Request__Validator
    sequence_runner      : Sequence__Runner
    action_runner        : Action__Runner
    credentials_loader   : Credentials__Loader

    def setup(self) -> 'Playwright__Service':                                       # Initialise sub-services
        self.capability_detector.detect()
        self.sequence_runner.setup(session_manager      = self.session_manager      ,
                                   browser_launcher     = self.browser_launcher     ,
                                   capability_detector  = self.capability_detector  ,
                                   request_validator    = self.request_validator    ,
                                   credentials_loader   = self.credentials_loader  )
        self.action_runner.setup(session_manager      = self.session_manager        ,
                                 capability_detector  = self.capability_detector    ,
                                 request_validator    = self.request_validator      )
        return self

    # ── Info / Health ──────────────────────────────────────────────────────────

    def get_service_info(self) -> Schema__Service__Info:
        return self.capability_detector.service_info()

    def get_health(self) -> Schema__Health:
        checks = [self.browser_launcher.healthcheck()          ,
                  self.session_manager.healthcheck()           ,
                  self.capability_detector.connectivity_check()]
        return Schema__Health(healthy   = all(c.healthy for c in checks) ,
                              checks    = checks                        ,
                              timestamp = Timestamp_Now()               )

    def get_capabilities(self) -> Schema__Service__Capabilities:
        return self.capability_detector.capabilities()

    # ── Session Lifecycle ──────────────────────────────────────────────────────

    def session_create(self, request: Schema__Session__Create__Request     ) -> Schema__Session__Create__Response:
        capabilities = self.capability_detector.capabilities()
        self.request_validator.validate_session_create(request, capabilities)
        trace_id = request.trace_id or Safe_Str__Trace_Id(self.generate_trace_id())
        browser  = self.browser_launcher.launch(request.browser_config)
        session  = self.session_manager.create(browser      = browser       ,
                                               request      = request       ,
                                               trace_id     = trace_id      ,
                                               capabilities = capabilities  )
        if request.credentials:
            self.credentials_loader.apply(session.session_id, self.session_manager, request.credentials)
        return Schema__Session__Create__Response(session_info = session      ,
                                                 capabilities = capabilities)

    def session_list(self) -> List[Schema__Session__Info]:
        return self.session_manager.list_active()

    def session_get(self, session_id: Session_Id) -> Schema__Session__Info:
        return self.session_manager.get(session_id)

    def session_save_state(self                                              ,
                           session_id : Session_Id                           ,
                           request    : Schema__Session__State__Save__Request
                      ) -> Schema__Session__State__Save__Response:
        session = self.session_manager.get(session_id)
        if session is None:
            return None
        browser = self.session_manager.get_browser(session_id)
        state   = browser.contexts()[0].storage_state()
        self.credentials_loader.save_state_to_vault(request.vault_ref, state)
        return Schema__Session__State__Save__Response(session_id = session_id         ,
                                                      vault_ref  = request.vault_ref  ,
                                                      saved_at   = Timestamp_Now()    )

    def session_close(self, session_id: Session_Id) -> Schema__Session__Close__Response:
        session = self.session_manager.get(session_id)
        if session is None:
            return None
        artefacts   = self.session_manager.get_artefacts(session_id)
        start_time  = self.session_manager.get_start_time(session_id)
        duration_ms = int((time.time() * 1000) - start_time)
        self.session_manager.close(session_id)
        self.browser_launcher.stop(session_id)
        return Schema__Session__Close__Response(session_info      = session      ,
                                                artefacts         = artefacts    ,
                                                total_duration_ms = duration_ms  )

    # ── Action + Sequence Execution (delegation) ───────────────────────────────

    def execute_action(self, request: Schema__Action__Request) -> Schema__Action__Response:
        return self.action_runner.execute(request)

    def sequence_execute(self, request: Schema__Sequence__Request) -> Schema__Sequence__Response:
        return self.sequence_runner.execute(request, self)                           # Pass self so runner can call session_create/close

    # ── Utility ────────────────────────────────────────────────────────────────

    def generate_trace_id(self) -> str:
        import uuid
        return uuid.uuid4().hex[:8]
```

### 4.2 Request__Validator

All validation logic extracted from the orchestrator.

```python
# ═══════════════════════════════════════════════════════════════════════════════
# Request__Validator — cross-schema validation
#
# Validates session-create requests, per-action capture config, step ID
# uniqueness, and evaluate allowlist. Raises HTTPException on failure.
# ═══════════════════════════════════════════════════════════════════════════════

from fastapi import HTTPException


class Request__Validator(Type_Safe):

    js_allowlist : JS__Expression__Allowlist

    def validate_session_create(self                                          ,      # Session-create cross-schema checks
                                request      : Schema__Session__Create__Request,
                                capabilities : Schema__Service__Capabilities
                           ) -> None:
        if request.lifetime_hint == Enum__Session__Lifetime.PERSISTENT_DISTRIBUTED:
            self.reject('distributed_not_supported',
                        'Persistent distributed sessions are not supported')

        if request.lifetime_hint == Enum__Session__Lifetime.PERSISTENT_SINGLE:
            if not capabilities.supports_persistent:
                self.reject('persistent_not_supported',
                            'Persistent sessions not available on this deployment',
                            capabilities=capabilities)

        cfg = request.browser_config
        if cfg.provider == Enum__Browser__Provider.CDP_CONNECT and not cfg.cdp_endpoint_url:
            self.reject('cdp_missing_endpoint',
                        'cdp_endpoint_url required when provider is cdp_connect')

    def validate_step(self                                                    ,     # Per-action capture + security checks
                      step         : Schema__Step__Base                       ,
                      cap_config   : Schema__Capture__Config                  ,
                      capabilities : Schema__Service__Capabilities            ,
                      target       : Enum__Deployment__Target
                 ) -> None:
        if step.action == Enum__Step__Action.EVALUATE:
            if not self.js_allowlist.is_allowed(step.expression):
                self.reject('evaluate_expression_not_allowed',
                            'JS expression not in trusted allowlist')

        self.validate_sink_configs(cap_config, capabilities, target)

    def validate_sink_configs(self                                            ,     # Check all enabled sinks against capabilities
                              cap_config   : Schema__Capture__Config          ,
                              capabilities : Schema__Service__Capabilities    ,
                              target       : Enum__Deployment__Target
                         ) -> None:
        sink_fields = ['screenshot', 'screenshot_on_fail', 'video', 'pdf',
                       'har', 'trace', 'console_log', 'network_log', 'page_content']
        for field_name in sink_fields:
            sink_config = getattr(cap_config, field_name)
            if not sink_config.enabled:
                continue
            if sink_config.sink not in capabilities.supported_sinks:
                self.reject('sink_incompatible_with_deployment',
                            f"Artefact sink '{sink_config.sink.value}' not available on "
                            f"deployment '{target.value}'. Supported: "
                            f"{[s.value for s in capabilities.supported_sinks]}",
                            capabilities=capabilities)
            if sink_config.sink == Enum__Artefact__Sink.VAULT and not sink_config.sink_vault_ref:
                self.reject('sink_missing_vault_ref',
                            f"sink_vault_ref required for '{field_name}' when sink=vault")
            if sink_config.sink == Enum__Artefact__Sink.S3 and not sink_config.sink_s3_bucket:
                self.reject('sink_missing_s3_bucket',
                            f"sink_s3_bucket required for '{field_name}' when sink=s3")

    def validate_step_ids_unique(self, steps: list) -> None:
        ids = [str(s.id) for s in steps if s.id]
        if len(ids) != len(set(ids)):
            dupes = list(set(i for i in ids if ids.count(i) > 1))
            self.reject('duplicate_step_ids', f'Duplicate step IDs: {dupes}')

    def validate_sequence_timeout(self                                       ,
                                  total_timeout_ms : int                     ,
                                  capabilities     : Schema__Service__Capabilities
                             ) -> None:
        if total_timeout_ms > capabilities.max_session_lifetime_ms:
            self.reject('timeout_exceeds_deployment_limit',
                        f'Requested {total_timeout_ms}ms exceeds deployment max '
                        f'{capabilities.max_session_lifetime_ms}ms',
                        capabilities=capabilities)

    def reject(self                                                          ,      # Uniform error response builder
               error_code   : str                                            ,
               error_message: str                                            ,
               capabilities : Schema__Service__Capabilities = None
          ) -> None:
        raise HTTPException(422, Schema__Error__Response(
            error_code   = error_code                                        ,
            error_message = error_message                                    ,
            capabilities  = capabilities                                     ,
        ).json())
```

### 4.3 Action__Runner (Layer 0 Execution)

```python
# ═══════════════════════════════════════════════════════════════════════════════
# Action__Runner — Layer 0 single-action execution
#
# Handles POST /browser/* requests. Looks up session, parses the step,
# validates, executes, records, returns typed response.
# ═══════════════════════════════════════════════════════════════════════════════


class Action__Runner(Type_Safe):

    session_manager      : Session__Manager       = None
    capability_detector  : Capability__Detector    = None
    request_validator    : Request__Validator      = None
    sequence_dispatcher  : Sequence__Dispatcher
    artefact_writer      : Artefact__Writer

    def setup(self                                                           ,
              session_manager     : Session__Manager                         ,
              capability_detector : Capability__Detector                     ,
              request_validator   : Request__Validator
         ) -> 'Action__Runner':
        self.session_manager     = session_manager
        self.capability_detector = capability_detector
        self.request_validator   = request_validator
        return self

    def execute(self, request: Schema__Action__Request) -> Schema__Action__Response:
        session = self.session_manager.get(request.session_id)
        if session is None:
            raise HTTPException(404, f"Session {request.session_id} not found")

        trace_id     = request.trace_id or session.trace_id
        browser      = self.session_manager.get_browser(request.session_id)
        page         = browser.contexts()[0].pages[0]
        cap_config   = request.capture_config or self.session_manager.get_capture_config(request.session_id)
        capabilities = self.capability_detector.capabilities()
        target       = self.capability_detector.target()

        step = self.sequence_dispatcher.parse_single_step(request.step, 0)
        self.request_validator.validate_step(step, cap_config, capabilities, target)

        step_result = self.sequence_dispatcher.execute_step(step            = step          ,
                                                            page            = page          ,
                                                            capture_config  = cap_config    ,
                                                            capabilities    = capabilities  ,
                                                            artefact_writer = self.artefact_writer)

        self.session_manager.record_action(request.session_id, step_result)
        session = self.session_manager.get(request.session_id)

        return Schema__Action__Response(session_id   = request.session_id ,
                                         trace_id    = trace_id           ,
                                         step_result = step_result        ,
                                         session_info = session           )
```

### 4.4 Sequence__Runner (Layer 3 Execution)

```python
# ═══════════════════════════════════════════════════════════════════════════════
# Sequence__Runner — Layer 3 sequence execution loop
#
# Parses all steps, runs them in order, handles halt-on-error,
# total timeout, skipping, and result aggregation.
# ═══════════════════════════════════════════════════════════════════════════════

import time


class Sequence__Runner(Type_Safe):

    session_manager      : Session__Manager       = None
    browser_launcher     : Browser__Launcher      = None
    capability_detector  : Capability__Detector    = None
    request_validator    : Request__Validator      = None
    credentials_loader   : Credentials__Loader    = None
    sequence_dispatcher  : Sequence__Dispatcher
    artefact_writer      : Artefact__Writer

    def setup(self                                                           ,
              session_manager     : Session__Manager                         ,
              browser_launcher    : Browser__Launcher                        ,
              capability_detector : Capability__Detector                     ,
              request_validator   : Request__Validator                       ,
              credentials_loader  : Credentials__Loader
         ) -> 'Sequence__Runner':
        self.session_manager     = session_manager
        self.browser_launcher    = browser_launcher
        self.capability_detector = capability_detector
        self.request_validator   = request_validator
        self.credentials_loader  = credentials_loader
        return self

    def execute(self                                                         ,
                request    : Schema__Sequence__Request                       ,
                service    : 'Playwright__Service'                           # For session_create / session_close delegation
           ) -> Schema__Sequence__Response:
        capabilities = self.capability_detector.capabilities()
        target       = self.capability_detector.target()
        trace_id     = request.trace_id or Safe_Str__Trace_Id(service.generate_trace_id())
        sequence_id  = request.sequence_id or Sequence_Id()

        self.request_validator.validate_sequence_timeout(request.sequence_config.total_timeout_ms, capabilities)

        # ── Resolve or create session ──────────────────────────────────────
        owns_session = False
        if request.session_id:
            session = self.session_manager.get(request.session_id)
            if session is None:
                raise HTTPException(404, f"Session {request.session_id} not found")
        else:
            create_req  = Schema__Session__Create__Request(
                              browser_config = request.browser_config or Schema__Browser__Config() ,
                              credentials    = request.credentials                                 ,
                              capture_config = request.capture_config                               ,
                              trace_id       = trace_id                                            )
            create_resp = service.session_create(create_req)
            session     = create_resp.session_info
            owns_session = True

        session_id = session.session_id
        browser    = self.session_manager.get_browser(session_id)
        page       = browser.contexts()[0].pages[0]
        cap_config = request.capture_config

        # ── Parse + validate steps ─────────────────────────────────────────
        parsed_steps = self.sequence_dispatcher.parse_steps(request.steps)
        self.request_validator.validate_step_ids_unique(parsed_steps)

        # ── Execute steps ──────────────────────────────────────────────────
        step_results = []
        total_start  = time.time()
        halt         = False

        for index, step in enumerate(parsed_steps):
            if halt:
                step_results.append(self.make_skipped_result(step, index))
                continue

            self.request_validator.validate_step(step, cap_config, capabilities, target)

            step_result            = self.sequence_dispatcher.execute_step(
                                         step            = step            ,
                                         page            = page            ,
                                         capture_config  = cap_config      ,
                                         capabilities    = capabilities    ,
                                         artefact_writer = self.artefact_writer)
            step_result.step_index = index                                          # Set correct index
            step_results.append(step_result)

            self.session_manager.record_action(session_id, step_result)

            if step_result.status == Enum__Step__Status.FAILED:
                if not step.continue_on_error and request.sequence_config.halt_on_error:
                    halt = True

            elapsed_ms = int((time.time() - total_start) * 1000)
            if elapsed_ms > request.sequence_config.total_timeout_ms:
                halt = True

        # ── Build response ─────────────────────────────────────────────────
        total_duration = int((time.time() - total_start) * 1000)
        passed         = sum(1 for r in step_results if r.status == Enum__Step__Status.PASSED )
        failed         = sum(1 for r in step_results if r.status == Enum__Step__Status.FAILED )
        skipped        = sum(1 for r in step_results if r.status == Enum__Step__Status.SKIPPED)

        if   failed == 0 : seq_status = Enum__Sequence__Status.COMPLETED
        elif halt         : seq_status = Enum__Sequence__Status.FAILED
        else              : seq_status = Enum__Sequence__Status.PARTIAL

        all_artefacts = []
        for r in step_results:
            all_artefacts.extend(r.artefacts)

        if request.close_session_after and owns_session:
            service.session_close(session_id)

        session = self.session_manager.get(session_id) or session

        return Schema__Sequence__Response(sequence_id       = sequence_id      ,
                                          trace_id          = trace_id         ,
                                          status            = seq_status       ,
                                          total_duration_ms = total_duration   ,
                                          steps_total       = len(parsed_steps),
                                          steps_passed      = passed           ,
                                          steps_failed      = failed           ,
                                          steps_skipped     = skipped          ,
                                          step_results      = step_results     ,
                                          session_info      = session          ,
                                          artefacts         = all_artefacts    )

    def make_skipped_result(self, step: Schema__Step__Base, index: int) -> Schema__Step__Result__Base:
        return Schema__Step__Result__Base(step_id     = step.id or Step_Id(str(index))  ,
                                          step_index  = index                           ,
                                          action      = step.action                     ,
                                          status      = Enum__Step__Status.SKIPPED      ,
                                          duration_ms = 0                               )
```

### 4.6 Capability__Detector (Type_Safe capabilities, no dicts)

```python
# ═══════════════════════════════════════════════════════════════════════════════
# Capability__Detector — detect deployment target and build capabilities

                path = pw.chromium.executable_path

        except Exception:
            return 'unknown'

        vault_url = get_env(ENV_VAR__SG_SEND_BASE_URL)
        return Schema__Health__Check(check_name = 'connectivity'             ,
                                     healthy    = bool(vault_url)           ,
                                     detail     = f'vault_url={vault_url}'  )
```

### 4.7 Credentials__Loader

```python
# ═══════════════════════════════════════════════════════════════════════════════
# Credentials__Loader — vault credential loading and saving
# ═══════════════════════════════════════════════════════════════════════════════


class Credentials__Loader(Type_Safe):

    artefact_writer : Artefact__Writer                                              # For vault read/write

    def apply(self                                                           ,
              session_id      : Session_Id                                   ,
              session_manager : Session__Manager                             ,
              credentials     : Schema__Session__Credentials
         ) -> None:
        browser = session_manager.get_browser(session_id)
        if browser is None:
            return
        context = browser.contexts()[0]

        if credentials.cookies_vault_ref:
            cookies = self.artefact_writer.read_from_vault(credentials.cookies_vault_ref)
            if cookies:
                context.add_cookies(cookies)

        if credentials.storage_state_vault_ref:
            state = self.artefact_writer.read_from_vault(credentials.storage_state_vault_ref)
            if state and 'cookies' in state:
                context.add_cookies(state['cookies'])

        if credentials.extra_http_headers:
            headers = {str(k): str(v) for k, v in credentials.extra_http_headers.items()}
            context.set_extra_http_headers(headers)

    def save_state_to_vault(self, vault_ref: Schema__Vault_Ref, state: dict) -> None:
        self.artefact_writer.write_to_vault(vault_ref, state)
```
