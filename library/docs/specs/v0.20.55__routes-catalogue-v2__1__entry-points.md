## Section 2 — Entry Points

### 2.1 Service Class (pure class definition — no module-level execution)

```python
# ═══════════════════════════════════════════════════════════════════════════════
# Fast_API__Playwright__Service — unified Playwright service class
#
# Pure class definition. Module-level setup lives in lambda_handler.py.
# ═══════════════════════════════════════════════════════════════════════════════

from osbot_fast_api_serverless.fast_api.Serverless__Fast_API import Serverless__Fast_API
from osbot_fast_api_serverless.fast_api.routes.Routes__Info  import Routes__Info


class Fast_API__Playwright__Service(Serverless__Fast_API):

    playwright_service : Playwright__Service = None             # Injected or auto-created in setup()

    def setup(self):
        with self.config as _:
            _.name           = 'SG/Playwright Service'
            _.version        = version__sgraph_ai_service_playwright
            _.description    = 'Browser automation service for SG/Send — runs everywhere'
            _.enable_api_key = False                            # Internal tooling; per-route auth later if needed

        if self.playwright_service is None:                     # If-None guard: allows test injection
            self.playwright_service = Playwright__Service()
            self.playwright_service.setup()

        return super().setup()

    def setup_routes(self):
        service = self.playwright_service
        self.add_routes(Routes__Info                                          )
        self.add_routes(Routes__Health    , playwright_service = service      )
        self.add_routes(Routes__Session   , playwright_service = service      )
        self.add_routes(Routes__Browser   , playwright_service = service      )
        self.add_routes(Routes__Sequence  , playwright_service = service      )
        return self
```

### 2.2 Lambda Handler (separate file — triggers process startup)

```python
# ═══════════════════════════════════════════════════════════════════════════════
# lambda_handler.py — module-level singletons for Lambda / uvicorn
#
# This file is the execution entry point. Importing it fires up the entire
# FastAPI app, Playwright service, and capability detection.
#
# Lambda Web Adapter:
#   CMD exec python3 sgraph_ai_service_playwright/fast_api/lambda_handler.py
#
# uvicorn (local dev):
#   uvicorn sgraph_ai_service_playwright.fast_api.lambda_handler:app --reload --port 8000
#
# Lambda direct invoke:
#   HANDLER=sgraph_ai_service_playwright.fast_api.lambda_handler.run
# ═══════════════════════════════════════════════════════════════════════════════

import sys
sys.path.append('/opt/python')                                  # LWA loses the path to Lambda layers

from sgraph_ai_service_playwright.fast_api.Fast_API__Playwright__Service import Fast_API__Playwright__Service


_fast_api = Fast_API__Playwright__Service().setup()

handler   = _fast_api.handler()                                 # Mangum-wrapped ASGI — for Lambda direct invoke
app       = _fast_api.app()                                     # Raw ASGI app — for uvicorn / Lambda Web Adapter


def run(event, context=None):                                   # Lambda entry point
    return handler(event, context)


if __name__ == '__main__':                                      # For: CMD exec python3 handler.py
    import uvicorn
    uvicorn.run(app, host='0.0.0.0', port=8000)
```

**Why separate files:** `Fast_API__Playwright__Service.py` is a class definition that can be imported without side effects (critical for test injection via `register_playwright_service__in_memory()`). `lambda_handler.py` is the process entry point that instantiates the singleton, starts Playwright, and binds to a port. Importing `lambda_handler` fires everything up — importing the class does not.
