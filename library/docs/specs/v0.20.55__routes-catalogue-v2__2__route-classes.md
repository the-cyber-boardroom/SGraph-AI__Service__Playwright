## Section 3 — Route Classes

### 3.1 Routes__Health

```python
# ═══════════════════════════════════════════════════════════════════════════════
# Routes__Health — service info and health endpoints
# GET /health/info          → Schema__Service__Info
# GET /health/status        → Schema__Health
# GET /health/capabilities  → Schema__Service__Capabilities
# ═══════════════════════════════════════════════════════════════════════════════

TAG__ROUTES_HEALTH = 'health'
ROUTES_PATHS__HEALTH = [
    f'/{TAG__ROUTES_HEALTH}/info'         ,
    f'/{TAG__ROUTES_HEALTH}/status'       ,
    f'/{TAG__ROUTES_HEALTH}/capabilities' ,
]


class Routes__Health(Fast_API__Routes):
    tag                : str                  = TAG__ROUTES_HEALTH
    playwright_service : Playwright__Service

    def info(self) -> dict:                                     # GET /health/info
        return self.playwright_service.get_service_info().json()

    def status(self) -> dict:                                   # GET /health/status
        return self.playwright_service.get_health().json()

    def capabilities(self) -> dict:                             # GET /health/capabilities
        return self.playwright_service.get_capabilities().json()

    def setup_routes(self):
        self.add_route_get(self.info        )
        self.add_route_get(self.status      )
        self.add_route_get(self.capabilities)
        return self
```

### 3.2 Routes__Session

**Note:** `save_state` now returns a proper Type_Safe schema, not a raw dict.

```python
# ═══════════════════════════════════════════════════════════════════════════════
# Routes__Session — session lifecycle endpoints
#
# POST   /session/create                → Schema__Session__Create__Response
# GET    /session/list                  → List[Schema__Session__Info]
# GET    /session/get/by-id/{id}        → Schema__Session__Info
# POST   /session/save-state/{id}       → Schema__Session__State__Save__Response
# DELETE /session/close/{id}            → Schema__Session__Close__Response
# ═══════════════════════════════════════════════════════════════════════════════

TAG__ROUTES_SESSION = 'session'
ROUTES_PATHS__SESSION = [
    f'/{TAG__ROUTES_SESSION}/create'               ,
    f'/{TAG__ROUTES_SESSION}/list'                  ,
    f'/{TAG__ROUTES_SESSION}/get/by-id/{{id}}'      ,
    f'/{TAG__ROUTES_SESSION}/save-state/{{id}}'     ,
    f'/{TAG__ROUTES_SESSION}/close/{{id}}'          ,
]


class Routes__Session(Fast_API__Routes):
    tag                : str                  = TAG__ROUTES_SESSION
    playwright_service : Playwright__Service

    def create(self, body: Schema__Session__Create__Request) -> dict:               # POST /session/create
        return self.playwright_service.session_create(body).json()

    def list(self) -> list:                                                         # GET /session/list
        return [s.json() for s in self.playwright_service.session_list()]

    def get__by_id__id(self, id: Session_Id) -> dict:                               # GET /session/get/by-id/{id}
        result = self.playwright_service.session_get(id)
        if result is None:
            raise HTTPException(status_code=404, detail=f"Session {id} not found")
        return result.json()

    def save_state__id(self                                                 ,       # Save session storage state to vault
                       id   : Session_Id                                    ,
                       body : Schema__Session__State__Save__Request
                  ) -> dict:                                                        # POST /session/save-state/{id}
        result = self.playwright_service.session_save_state(id, body)
        if result is None:
            raise HTTPException(status_code=404, detail=f"Session {id} not found")
        return result.json()

    def close__id(self, id: Session_Id) -> dict:                                    # DELETE /session/close/{id}
        result = self.playwright_service.session_close(id)
        if result is None:
            raise HTTPException(status_code=404, detail=f"Session {id} not found")
        return result.json()

    def setup_routes(self):
        self.add_route_post  (self.create        )
        self.add_route_get   (self.list           )
        self.add_route_get   (self.get__by_id__id )
        self.add_route_post  (self.save_state__id )
        self.add_route_delete(self.close__id      )
        return self
```

**New schema needed (addendum to schema catalogue v2):**

```python
class Schema__Session__State__Save__Response(Type_Safe):                            # POST /session/{id}/save-state response
    session_id          : Session_Id
    vault_ref           : Schema__Vault_Ref                                         # Where the state was written
    saved_at            : Timestamp_Now
```

### 3.3 Routes__Browser (Layer 0 — Direct Actions)

```python
# ═══════════════════════════════════════════════════════════════════════════════
# Routes__Browser — Layer 0 direct browser actions
#
# Every route is POST. Each delegates to Action__Runner.execute() which:
#   1. Looks up the session
#   2. Parses the step dict into the correct Schema__Step__* type
#   3. Validates capture config against current capabilities
#   4. Executes via Step__Executor
#   5. Writes artefacts
#   6. Returns Schema__Action__Response
# ═══════════════════════════════════════════════════════════════════════════════

TAG__ROUTES_BROWSER = 'browser'
ROUTES_PATHS__BROWSER = [
    f'/{TAG__ROUTES_BROWSER}/navigate'       ,
    f'/{TAG__ROUTES_BROWSER}/click'          ,
    f'/{TAG__ROUTES_BROWSER}/fill'           ,
    f'/{TAG__ROUTES_BROWSER}/press'          ,
    f'/{TAG__ROUTES_BROWSER}/select'         ,
    f'/{TAG__ROUTES_BROWSER}/hover'          ,
    f'/{TAG__ROUTES_BROWSER}/scroll'         ,
    f'/{TAG__ROUTES_BROWSER}/wait-for'       ,
    f'/{TAG__ROUTES_BROWSER}/screenshot'     ,
    f'/{TAG__ROUTES_BROWSER}/video-start'    ,
    f'/{TAG__ROUTES_BROWSER}/video-stop'     ,
    f'/{TAG__ROUTES_BROWSER}/evaluate'       ,
    f'/{TAG__ROUTES_BROWSER}/dispatch-event' ,
    f'/{TAG__ROUTES_BROWSER}/set-viewport'   ,
    f'/{TAG__ROUTES_BROWSER}/get-content'    ,
    f'/{TAG__ROUTES_BROWSER}/get-url'        ,
]

    def fill(self, body: Schema__Action__Request) -> dict:                          # POST /browser/fill
        return self.playwright_service.execute_action(body).json()

    def press(self, body: Schema__Action__Request) -> dict:                         # POST /browser/press
        return self.playwright_service.execute_action(body).json()

    def select(self, body: Schema__Action__Request) -> dict:                        # POST /browser/select
        return self.playwright_service.execute_action(body).json()

    def hover(self, body: Schema__Action__Request) -> dict:                         # POST /browser/hover
        return self.playwright_service.execute_action(body).json()

    def scroll(self, body: Schema__Action__Request) -> dict:                        # POST /browser/scroll
        return self.playwright_service.execute_action(body).json()

    def wait_for(self, body: Schema__Action__Request) -> dict:                      # POST /browser/wait-for
        return self.playwright_service.execute_action(body).json()

    def screenshot(self, body: Schema__Action__Request) -> dict:                    # POST /browser/screenshot
        return self.playwright_service.execute_action(body).json()

        return self.playwright_service.execute_action(body).json()

        return self.playwright_service.execute_action(body).json()
```

### 3.4 Routes__Sequence (Layer 3 — Batch Execution)

```python
# ═══════════════════════════════════════════════════════════════════════════════
# Routes__Sequence — Layer 3 sequence execution
# POST /sequence/execute    → Schema__Sequence__Response
# ═══════════════════════════════════════════════════════════════════════════════

TAG__ROUTES_SEQUENCE = 'sequence'
ROUTES_PATHS__SEQUENCE = [
    f'/{TAG__ROUTES_SEQUENCE}/execute' ,
]


class Routes__Sequence(Fast_API__Routes):
    tag                : str                  = TAG__ROUTES_SEQUENCE
    playwright_service : Playwright__Service

    def execute(self, body: Schema__Sequence__Request) -> dict:                     # POST /sequence/execute
        return self.playwright_service.sequence_execute(body).json()

    def setup_routes(self):
        self.add_route_post(self.execute)
        return self
```
